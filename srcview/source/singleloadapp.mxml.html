<?xml version="1.0" encoding="utf-8"?>
<mx:Application xmlns:mx="http://www.adobe.com/2006/mxml"
                viewSourceURL="srcview/index.html">
    
    <mx:Script>
        <![CDATA[
            import com.swfloader.SWFItem;
            import com.swfloader.SWFLoaderManager;
            
            import components.IComponent;
            
            protected function button1_clickHandler(event:MouseEvent):void
            {
                var item:SWFItem = new SWFItem();
                item.url = "components/MyButtonComp.swf";
                item.completeHandler = function():void {
                    panel1.addChild(IComponent(item.content).getInstance());
                };
                
                SWFLoaderManager.load(item);
            }
        ]]>
    </mx:Script>
    
    
    <mx:Panel id="panel1" title="Single SWF Load" 
              height="300" width="400" 
              paddingTop="10" paddingLeft="10" paddingRight="10" paddingBottom="10">
        <mx:Button label="Load" click="button1_clickHandler(event)"/>
    </mx:Panel>
</mx:Application>
