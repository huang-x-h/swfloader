<?xml version="1.0" encoding="utf-8"?>
<mx:Application xmlns:mx="http://www.adobe.com/2006/mxml"
                viewSourceURL="srcview/index.html">
    
    <mx:Script>
        <![CDATA[
            import com.swfloader.SWFListLoader;
            
            protected function button2_clickHandler(event:MouseEvent):void
            {
                var loader:SWFListLoader = new SWFListLoader(["module/MyTextComp.swf", "module/MyLoginComp.swf"], function():void {
                    var textClazz:Class = forName("components.MyText");
                    var loginClazz:Class = forName("components.MyLogin");
                    
                    panel2.addChild(new textClazz());
                    panel2.addChild(new loginClazz());
                });
                
                loader.load();
            }
            
            public function forName(clazzName:String, applicationDomain:ApplicationDomain=null):Class
            {
                applicationDomain = applicationDomain == null ? ApplicationDomain.currentDomain : applicationDomain;
                
                var result:Class;
                if (applicationDomain.hasDefinition(clazzName))
                    result = applicationDomain.getDefinition(clazzName) as Class;
                return result;
            }
        ]]>
    </mx:Script>
    
    <mx:Panel id="panel2" title="Mutiple SWF Load" 
              height="300" width="400" 
              paddingTop="10" paddingLeft="10" paddingRight="10" paddingBottom="10">
        <mx:Button label="Load" click="button2_clickHandler(event)"/>
    </mx:Panel>
    
</mx:Application>
