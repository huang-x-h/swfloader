<?xml version="1.0" encoding="utf-8"?>
<mx:Application xmlns:mx="http://www.adobe.com/2006/mxml"
                viewSourceURL="srcview/index.html">
    
    <mx:Script>
        <![CDATA[
            import com.swfloader.SWFCache;
            import com.swfloader.SWFListLoader;
            
            import components.IComponent;
            
            private var swfCache:SWFCache = SWFCache.getInstance();
            
            protected function button2_clickHandler(event:MouseEvent):void
            {
                var loader:SWFListLoader = new SWFListLoader(["components/MyTextComp.swf", "components/MyLoginComp.swf"], function():void {
                    panel2.addChild(IComponent(swfCache.get("components/MyTextComp.swf")).getInstance());
                    panel2.addChild(IComponent(swfCache.get("components/MyLoginComp.swf")).getInstance());
                });
                
                loader.load();
            }
        ]]>
    </mx:Script>
    
    <mx:Panel id="panel2" title="Mutiple SWF Load" 
              height="300" width="400" 
              paddingTop="10" paddingLeft="10" paddingRight="10" paddingBottom="10">
        <mx:Button label="Load" click="button2_clickHandler(event)"/>
    </mx:Panel>
    
</mx:Application>
